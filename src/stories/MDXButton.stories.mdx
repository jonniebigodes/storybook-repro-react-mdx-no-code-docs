import { Canvas, Meta, Story } from "@storybook/addon-docs";

import { Button } from "./Button";

<Meta title="MDX/ButtonExample" component={Button} />

## With Default render function

<Canvas>
  <Story
    name="Button with render function"
    args={{

      label: "Boop",
    }}

  />

## Example without render function

  </div>
);
